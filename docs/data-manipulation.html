df_long= gather(df,Mes_type,length,1:4)
head(df_long)
##   Species     Mes_type length
## 1  setosa Sepal.Length    5.1

#grouping and summarizing data
# group by
df_sumzd=group_by(df_long,Species)

## # A tibble: 6 x 3
